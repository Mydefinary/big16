   ```nginx
   location /question {
     alias /usr/share/nginx/html;
     
     # 정적 파일 캐시 (중첩 location)
     location ~* \.(js|css|png|jpg|jpeg|gif|ico|svg)$ {
       expires 1y;
       add_header Cache-Control "public, immutable";
       try_files $uri =404;
     }
     
     # SPA fallback
     try_files $uri /question/index.html;
   }
   ```
